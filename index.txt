            </div>

                <label class="block text-xs font-bold text-gray-500 uppercase mb-1">Mensaje Personalizado</label>

            <button onclick="generateAll()"

                GENERAR HERRAMIENTAS

            <div id="result-container" class="hidden space-y-6 pt-4 border-t border-gray-100">
                
                <div>

                    <div class="flex gap-2">
                        <input type="text" id="final-link" readonly

                        <button onclick="copyToClipboard()" class="bg-green-100 text-green-700 px-4 py-2 rounded-lg font-bold text-sm hover:bg-green-200">
                            Copiar

    <script>
        let qrcode = null;

        function generateAll() {
            const phone = document.getElementById('phone').value;
            const message = encodeURIComponent(document.getElementById('message').value);
            
            if (!phone) {
                alert("Por favor, ingresa un número de teléfono.");
                return;
            }

            const url = `https://wa.me/${phone}?text=${message}`;
            
            // Mostrar Link
            document.getElementById('final-link').value = url;
            document.getElementById('result-container').classList.remove('hidden');

            // Generar QR
            const qrContainer = document.getElementById("qrcode");
            qrContainer.innerHTML = ""; // Limpiar QR anterior
            
            new QRCode(qrContainer, {
                text: url,
                width: 180,
                height: 180,
                colorDark : "#000000",
                colorLight : "#ffffff",
                correctLevel : QRCode.CorrectLevel.H
            });
        }

        function copyToClipboard() {
            const copyText = document.getElementById("final-link");
            copyText.select();
            document.setSelectionRange(0, 99999);
            navigator.clipboard.writeText(copyText.value);
            alert("¡Enlace copiado!");
        }
    </script>
